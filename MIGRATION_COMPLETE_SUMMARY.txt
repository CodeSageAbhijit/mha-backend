🎓 QUERY TRANSLATION CHEAT SHEET

FIND ONE:
  OLD: await collection.find_one({"email": "john@example.com"})
  NEW: db.query(User).filter(User.email == "john@example.com").first()

FIND ALL:
  OLD: await collection.find({}).to_list(None)
  NEW: db.query(User).all()

INSERT:
  OLD: await collection.insert_one({"name": "John", "email": "john@example.com"})
  NEW: user = User(id=uuid, name="John", email="john@example.com")
       db.add(user)
       db.commit()

UPDATE:
  OLD: await collection.update_one({"_id": ObjectId(id)}, {"$set": {"name": "Jane"}})
  NEW: user = db.query(User).filter(User.id == id).first()
       user.name = "Jane"
       db.commit()

DELETE:
  OLD: await collection.delete_one({"_id": ObjectId(id)})
  NEW: user = db.query(User).filter(User.id == id).first()
       db.delete(user)
       db.commit()
